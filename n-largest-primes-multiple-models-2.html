import math

def sieve_of_eratosthenes(start, end):
    primes = [True] * (end+1)
    p = 2
    while p*p <= end:
        if primes[p]:
            for i in range(p*p, end+1, p):
                primes[i] = False
        p += 1
    return [p for p in range(start, end) if primes[p]]

def n_largest_primes(start, end, N):
    if start > end:
        start, end = end, start
    primes = sieve_of_eratosthenes(start, end)
    return sorted(primes, reverse=True)[:N]
