def is_prime(n):
    if n < 2:
        return False
    for i in range(2, int(n ** 0.5) + 1):
        if n % i == 0:
            return False
    return True

def largest_primes(start, end, count):
    primes = []
    current_count = 0

    for n in range(start, end + 1):
        if is_prime(n):
            primes.append(n)
            current_count += 1
            if current_count > count:
                break

    return sorted(primes, reverse=True)
